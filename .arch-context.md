```java
package io.b2mash.b2b.b2bstrawman.correspondence;

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.EnumType;
import jakarta.persistence.Enumerated;
import jakarta.persistence.GeneratedValue;
import jakarta.persistence.GenerationType;
import jakarta.persistence.Id;
import jakarta.persistence.PrePersist;
import jakarta.persistence.PreUpdate;
import jakarta.persistence.Table;
import jakarta.persistence.Version;
import java.time.Instant;
import java.util.List;
import java.util.UUID;
import org.hibernate.annotations.JdbcTypeCode;
import org.hibernate.type.SqlTypes;

/**
 * A single filed email/correspondence (one row per filed message). Linkage is flat with at least
 * one of {@code customerId} / {@code projectId} non-null (enforced by the table CHECK and by
 * service validation). Idempotency is keyed on {@code messageId} (UNIQUE index).
 *
 * <p>Pure schema-per-tenant isolation — no {@code tenant_id} column, no {@code @Filter}.
 */
@Entity
@Table(name = "correspondence")
public class Correspondence {

  @Id
  @GeneratedValue(strategy = GenerationType.UUID)
  private UUID id;

  @Column(name = "customer_id")
  private UUID customerId; // raw UUID FK, not @ManyToOne

  @Column(name = "project_id")
  private UUID projectId; // raw UUID FK (the matter)

  @Enumerated(EnumType.STRING)
  @Column(name = "direction", nullable = false, length = 10)
  private Direction direction = Direction.INBOUND;

  @Column(name = "subject", length = 500)
  private String subject;

  @Column(name = "body_text", columnDefinition = "TEXT")
  private String bodyText;

  @Column(name = "body_html", columnDefinition = "TEXT")
  private String bodyHtml;

  @Column(name = "from_address", nullable = false, length = 320)
  private String fromAddress;

  @JdbcTypeCode(SqlTypes.JSON)
  @Column(name = "to_addresses", columnDefinition = "jsonb")
  private List<String> toAddresses;

  @JdbcTypeCode(SqlTypes.JSON)
  @Column(name = "cc_addresses", columnDefinition = "jsonb")
  private List<String> ccAddresses;

  @Column(name = "sent_at")
  private Instant sentAt;

  @Column(name = "received_at")
  private Instant receivedAt;

  @Column(name = "thread_key", length = 255)
  private String threadKey;

  @Column(name = "message_id", nullable = false, length = 512)
  private String messageId; // idempotency key

  @Column(name = "source", nullable = false, length = 30)
  private String source = "MCP";

  @Column(name = "filed_by_member_id", nullable = false)
  private UUID filedByMemberId;

  @Column(name = "filed_at", nullable = false)
  private Instant filedAt;

  @Column(name = "created_at", nullable = false, updatable = false)
  private Instant createdAt;

  @Column(name = "updated_at", nullable = false)
  private Instant updatedAt;

  @Version private int version;

  /** JPA-required no-arg constructor. */
  protected Correspondence() {}

  /**
   * Construct an inbound correspondence ready to be filed. Timestamps are set by {@link
   * #onCreate()} — do not set them here.
   */
  public Correspondence(
      UUID customerId,
      UUID projectId,
      String subject,
      String bodyText,
      String bodyHtml,
      String fromAddress,
      List<String> toAddresses,
      List<String> ccAddresses,
      Instant sentAt,
      Instant receivedAt,
      String threadKey,
      String messageId,
      String source,
      UUID filedByMemberId) {
    this.customerId = customerId;
    this.projectId = projectId;
    this.direction = Direction.INBOUND;
    this.subject = subject;
    this.bodyText = bodyText;
    this.bodyHtml = bodyHtml;
    this.fromAddress = fromAddress;
    this.toAddresses = toAddresses;
    this.ccAddresses = ccAddresses;
    this.sentAt = sentAt;
    this.receivedAt = receivedAt;
    this.threadKey = threadKey;
    this.messageId = messageId;
    this.source = (source == null || source.isBlank()) ? "MCP" : source;
    this.filedByMemberId = filedByMemberId;
  }

  @PrePersist
  void onCreate() {
    var now = Instant.now();
    this.createdAt = now;
    this.updatedAt = now;
    if (this.filedAt == null) {
      this.filedAt = now;
    }
  }

  @PreUpdate
  void onUpdate() {
    this.updatedAt = Instant.now();
  }

  public UUID getId() { return id; }
  public UUID getCustomerId() { return customerId; }
  public UUID getProjectId() { return projectId; }
  public Direction getDirection() { return direction; }
  public String getSubject() { return subject; }
  public String getBodyText() { return bodyText; }
  public String getBodyHtml() { return bodyHtml; }
  public String getFromAddress() { return fromAddress; }
  public List<String> getToAddresses() { return toAddresses; }
  public List<String> getCcAddresses() { return ccAddresses; }
  public Instant getSentAt() { return sentAt; }
  public Instant getReceivedAt() { return receivedAt; }
  public String getThreadKey() { return threadKey; }
  public String getMessageId() { return messageId; }
  public String getSource() { return source; }
  public UUID getFiledByMemberId() { return filedByMemberId; }
  public Instant getFiledAt() { return filedAt; }
  public Instant getCreatedAt() { return createdAt; }
  public Instant getUpdatedAt() { return updatedAt; }
  public int getVersion() { return version; }
}
```

Key entity conventions: `@GeneratedValue(strategy = GenerationType.UUID)`; raw UUID FKs (not `@ManyToOne`) across aggregate boundaries; snake_case `@Column(name=…)` always explicit; `@PrePersist`/`@PreUpdate` timestamps; `@Version int version` for optimistic locking; jsonb via `@JdbcTypeCode(SqlTypes.JSON)`; no Lombok; protected no-arg ctor + full-arg public ctor; no tenant_id column (schema-per-tenant).

```markdown
# ADR-324: Correspondence Read-Back over MCP — Body-from-Kazi, Project-Access-Gated, Existing-Egress-Consent

**Status**: Accepted

**Context**:

Phase 81 made the Kazi MCP server able to **write** correspondence (`file_correspondence`, `attach_document`, `propose_task`, ADR-319/321/322): a firm's own Claude (BYOC) files inbound email into the right matter, attaches its documents, and proposes follow-up tasks for in-Kazi approval. It left a read/write asymmetry — **Claude can write correspondence into Kazi but cannot read it back over MCP.** The only correspondence read surface is the in-app REST "Correspondence" tab (Epic 586A), which is not an MCP tool. The motivating consumer is a `correspondence-digest` skill in `../claude-for-legal-sa` that answers "across my matters' inboxes, what needs a reply and what implies a deadline?" — and the deadlines and obligations live in the email **body**, not the subject line, so a useful digest must read the body.

The `correspondence/` bounded context already persists everything needed: the `Correspondence` entity holds `subject`, `bodyText`, `bodyHtml`, `fromAddress`, `toAddresses`, `ccAddresses`, `sentAt`, `receivedAt`, `threadKey`, `messageId`, `direction`, and linkage (`customerId`/`projectId`); `CorrespondenceService.listForProject`/`listForCustomer` already return `Page<CorrespondenceListResponse>` with view-access enforcement and a 50-row clamp. So the design question is **not** how to model correspondence — that exists — but how to expose a read of it over MCP: where the body comes from, how the read is gated, and what consent governs egressing the body. Three constraints frame it: a legal correspondence body is sensitive client PII (privileged material, deadlines); the in-app tab already enforces per-matter view-access; and the firm's existing read tools (`get_matter`, `get_client`) already egress comparable PII under a single read-egress consent.

**Options Considered**:

### A. Where the body comes from

1. **Read the body from Kazi via `get_correspondence` (CHOSEN)** — a new body-bearing read tool over the already-filed `Correspondence` row.
   - Pros: provider-neutral — works regardless of whether the firm uses Gmail, Outlook, or a pasted email; works for **archived/deleted** mail (Kazi holds the filed copy); identical egress posture to `get_matter`/`get_client` (one consent, one audit family); reuses the existing entity and `findById` (zero new storage); the body Claude reads is exactly the body Kazi filed (no drift).
   - Cons: requires Kazi to hold and egress the body (it already does — that is what `file_correspondence` persisted); the digest sees only correspondence that was filed (un-filed mail is invisible — acceptable: the digest is about *filed matter correspondence*).
2. **Re-hydrate the body from the firm's Gmail by `messageId`** — return only metadata over MCP; have Claude fetch the body from the Gmail connector.
   - Pros: Kazi never re-egresses the body; the body stays in the mail provider.
   - Cons: **provider-locks the digest to Gmail** (founder rejection) — Outlook/IMAP/pasted-email firms get no body; breaks for archived or deleted mail (Message-ID no longer resolvable); splits the read across two systems with two trust boundaries; the `messageId` is already in Kazi anyway. Rejected in the 2026-06-27 founder ideation precisely because it would make the digest Gmail-only.
3. **Metadata-only / no body at all** — ship `list_correspondence` and stop; never expose the body.
   - Pros: smallest surface; no body egress to reason about.
   - Cons: too thin — the digest's entire value (deadlines, obligations, "what needs a reply") lives in the body; a subject-only digest is not useful. Rejected.

### B. Gating model for the two read tools

1. **Project view-access, mirroring `get_matter`/`search_documents`/`get_document_url` (CHOSEN)** — `list_correspondence`'s matter path and `get_correspondence` resolve the correspondence's `projectId` and call `requireViewAccess`. `emitDenied(tool, "project-access", …)` is emitted **only** when the record **is found** with a `projectId` and `requireViewAccess` **refuses** — never for an absent or cross-tenant id, which is invisible to `findById` (`Optional.empty()` → `ResourceNotFoundException` → `not_found` with **no** `mcp.access.denied`, a lookup miss is not a policy denial). Both surface a non-leaking `not_found`. This mirrors `get_matter` exactly. Customer-only correspondence (no `projectId`) falls back to existence-in-tenant, matching `listForCustomer`/`search_documents` CUSTOMER scope.
   - Pros: **matches the in-app REST tab exactly** — an MCP user sees precisely the correspondence their Kazi role permits, no more; consistent with the sibling single-entity reads on related domains (matters, documents); reuses `CorrespondenceService.listForProject`'s existing `requireViewAccess`; correspondence is matter-scoped firm data of the same sensitivity as documents, so it should gate like documents.
   - Cons: the digest degrades for correspondence on matters the member cannot view (correct behaviour, but worth stating); two code paths (project vs customer-only) — minor, already the established `search_documents` shape.
2. **Org-wide `MCP_ACCESS` capability gate, like `resolve_matter_by_email`** — wrap both tools in `McpCapabilityGuard.gatedTool("MCP_ACCESS", …)`; any authenticated tenant member can read any correspondence.
   - Pros: simpler, org-uniform, one gate.
   - Cons: **less restrictive than the in-app UI** — a member could read correspondence (including the privileged body) on a matter they cannot see in the app; and the `MCP_ACCESS` front-door gate is not yet live (565B pending), so relying on it alone would leave the tools effectively ungated today. Wrong for body-bearing matter-scoped PII. Rejected.
3. **A new correspondence-specific read capability** — e.g. `MCP_READ_CORRESPONDENCE`.
   - Pros: granular opt-in for body read-back.
   - Cons: a new capability to provision, document, and test for no behavioural gain over project-access; correspondence is not more sensitive than documents (which use project-access, not a bespoke capability); over-engineering for a two-tool phase. Rejected.

### C. Consent posture for body egress

1. **Inherit the existing read-egress consent (CHOSEN)** — the body egresses under the same enablement + POPIA-egress-consent gate (`McpEnablementService.effectiveState()`) that already governs `get_matter`/`get_client`.
   - Pros: a correspondence body is the same POPIA category as the client PII `get_matter`/`get_client` already egress under that consent; one consent for the firm to reason about; no new flag to provision, surface in the UI, or migrate; revoking consent or disabling the connector turns body read-back off uniformly with every other read.
   - Cons: the firm cannot consent to matter/client reads but *withhold* correspondence-body reads (acceptable — same sensitivity class; a separate flag implies a distinction that does not exist).
2. **A new, stricter consent flag for bodies** — a dedicated "allow correspondence body egress" toggle.
   - Pros: lets a firm enable metadata reads but block body egress.
   - Cons: implies the body is a *new* POPIA category when it is the same class of client PII already egressed; new flag to build, default, migrate, and document; fragments the consent model. Rejected.
3. **No consent** — egress the body with only the enablement check.
   - Pros: none beyond the enablement gate already present.
   - Cons: egressing client PII to an external LLM with no POPIA consent is unacceptable. Rejected.

**Decision**: Add two MCP **read** tools on a new `CorrespondenceReadTools` bean. `list_correspondence(matterId | customerId, page)` returns the existing metadata projection as an `McpPage`, reusing `CorrespondenceService.listForProject`/`listForCustomer` verbatim (option **A1** for the read source, no new list path). On the customer path, `listForCustomer` does `customerRepository.findById(customerId).orElseThrow(ResourceNotFoundException)` before listing, so an unknown/cross-tenant `customerId` returns a non-leaking `not_found` (not a silent empty page, and no `mcp.access.denied`). `get_correspondence(id)` returns a single record **with body** via one net-new `CorrespondenceService.requireDetailById` method and a new `McpCorrespondenceDto` (the JPA entity never crosses the MCP boundary). Both are gated by **project view-access** (option **B1**) — `requireViewAccess(projectId)` when the correspondence is matter-scoped, existence-in-tenant when customer-only — surfacing a found-but-refused access denial as a non-leaking `not_found` + `mcp.access.denied(gate=project-access)`, while an absent/cross-tenant id surfaces as `not_found` with **no** `mcp.access.denied` (security-by-obscurity). This is **not** the org-wide `MCP_ACCESS` capability gate — and supersedes the requirements doc's `McpCapabilityGuard.gatedTool("MCP_ACCESS", …)` suggestion: the new read tools do **not** call `McpCapabilityGuard`, matching `get_matter`/`get_document_url`. Body egress rides the **existing** read-egress consent (option **C1**) — no new consent flag, no new gate, no new audit family. Both tools sit behind the enablement + POPIA-consent guard and emit the pre-existing `mcp.tool.invoked` read audit carrying **safe entity refs only** (correspondence id, matter/customer ids) — never subject, from-address, or body. **No new entity, table, migration, write tool, LLM, or Gmail integration.**

**Rationale**:

1. **Body from Kazi keeps the digest provider-neutral and durable.** Reading from the filed `Correspondence` row works for Gmail, Outlook, IMAP, and pasted email alike, and for archived/deleted mail — the firm's filed copy is the source of truth. Re-hydrating from Gmail would provider-lock the digest, which the founder rejected.
2. **Project-access matches the UI and the sibling reads.** Correspondence is matter-scoped firm data of the same sensitivity as documents; gating it like `get_matter`/`get_document_url` means an MCP user reads exactly what their Kazi role permits — no MCP-vs-UI asymmetry. An org-wide capability gate would be *looser* than the in-app tab and is not even live yet (565B).
3. **Inheriting read-egress consent is honest about the data class.** A correspondence body is the same POPIA category of client PII that `get_matter`/`get_client` already egress under one consent. A new flag would imply a distinction that does not exist and fragment the firm's consent model.
4. **Reuse over rebuild.** `list_correspondence` reuses the access-checked, clamped public `listForProject`/`listForCustomer` (**not** the lower-level package-private `listByProject`/`listByCustomer`, which bypass both the view-access check and the `McpPagination` page clamp — the requirements doc loosely names `listBy*`, but wiring those would skip the gate and the 50-row clamp); `get_correspondence` adds exactly one service method over the inherited `findById`. No new repository method, no new storage, no migration.
5. **Read capability, read audit, no PII in audit.** The tools emit only the pre-existing `mcp.tool.invoked` / `mcp.access.denied` (no new audit event type — `AuditEventTypeRegistry` stays at 36) with `entityRef`s only, mirroring `resolve_matter_by_email`'s "entityRefs carries ids only, never the email/name (POPIA)". The body is *returned to the LLM* by design but *never written to audit details*.

**Consequences**:
- Positive: closes the Phase 81 read/write asymmetry; the `correspondence-digest` skill can ground on filed correspondence including the body; provider-neutral and works for archived mail; one consent, one audit family, zero new POPIA category; no new entity/table/migration/write-path.
- Positive: an MCP user reads exactly the correspondence their Kazi role permits — identical to the in-app tab; tenant-isolation holds via search_path (a cross-tenant id returns `not_found`, never a body).
- Positive: the only registry-test edit is one line in `ALLOWED_TOOL_BEANS`; no audit-count bump; no regex/exemption changes (`list_`/`get_` already match the read-verb pattern).
- Negative: **the digest degrades for correspondence whose matter the member cannot view** — the member sees only what their Kazi role permits, exactly as with documents. This is correct behaviour, but it means a digest's completeness depends on the running member's matter access (a firm owner sees more than a member).
- Negative: a body-bearing read tool egresses privileged client PII to an external LLM — acceptable only because it inherits the firm's deliberate read-egress consent and the enablement gate; if the firm has not consented, both tools return `not_enabled` and egress nothing.
- Negative: customer-only correspondence (no `projectId`) is gated by existence-in-tenant rather than per-matter view-access — consistent with `listForCustomer`/`search_documents` CUSTOMER scope, but slightly looser than the matter path; the asymmetry is inherent to the data model (a correspondence with no matter has no matter to access-check).
- Related: [ADR-319](ADR-319-inbound-correspondence-domain.md) (the correspondence domain these tools read), [ADR-321](ADR-321-mcp-write-tool-category.md) (read/write capability split — these stay read-capability), [ADR-322](ADR-322-tiered-write-safety-and-gate-over-mcp.md) (Phase 81 write-safety context), [ADR-323](ADR-323-email-matter-linking.md) (closest read-tool precedent — argument discipline + POPIA-safe audit), ADR-304 (tenant-isolation/capability-gating model), ADR-306 (`McpPage` pagination envelope).
```

```java
/**
 * Time-tracking reminder settings group (Wave 3.5 embeddable refactor). Holds the firm's
 * daily-time-entry reminder configuration: the enabled toggle, the working-day CSV (e.g. {@code
 * "MON,TUE,WED,THU,FRI"}), the time-of-day the reminder fires, and the minimum logged minutes below
 * which a member is nudged. Consumed by the time-reminder scheduler/handler.
 *
 * <p>Persisted inline on the {@code org_settings} table via {@code @Embedded} +
 * {@code @AttributeOverride} on {@link OrgSettings}. Column names, types, and nullability are
 * UNCHANGED from when these fields lived directly on the entity (zero schema change — see {@code
 * OrgSettingsSchemaSnapshotTest}). {@code time_reminder_enabled} is NOT NULL, so it is modelled as
 * a primitive {@code boolean} (defaults to {@code false}) and the embedded never fully materialises
 * as NULL on reload; the remaining three columns are nullable. The {@code time_reminder_time}
 * column stays mapped to {@link LocalTime} ({@code time without time zone}).
 *
 * <p>Field-level setters here intentionally do NOT bump {@code OrgSettings.updatedAt}; the
 * embeddable has no reference to the owning entity's timestamp. Hibernate dirty-checks the embedded
 * columns and persists changes regardless. The {@link #updateTimeReminderSettings} domain mutator
 * preserves the same atomic four-field update semantics the entity method used to provide.
 */
@Embeddable
public class TimeReminderSettings {

  private static final Map<String, DayOfWeek> DAY_ABBREVIATIONS =
      Map.of(
          "MON", DayOfWeek.MONDAY,
          "TUE", DayOfWeek.TUESDAY,
          "WED", DayOfWeek.WEDNESDAY,
          "THU", DayOfWeek.THURSDAY,
          "FRI", DayOfWeek.FRIDAY,
          "SAT", DayOfWeek.SATURDAY,
          "SUN", DayOfWeek.SUNDAY);

  @Column(name = "time_reminder_enabled", nullable = false)
  private boolean timeReminderEnabled;

  @Column(name = "time_reminder_days", length = 50)
  private String timeReminderDays;

  @Column(name = "time_reminder_time")
  private LocalTime timeReminderTime;

  @Column(name = "time_reminder_min_minutes")
  private Integer timeReminderMinMinutes;

  protected TimeReminderSettings() {}

  public boolean isTimeReminderEnabled() { return timeReminderEnabled; }
  public void setTimeReminderEnabled(boolean timeReminderEnabled) { this.timeReminderEnabled = timeReminderEnabled; }
  public String getTimeReminderDays() { return timeReminderDays; }
  public void setTimeReminderDays(String timeReminderDays) { this.timeReminderDays = timeReminderDays; }
  public LocalTime getTimeReminderTime() { return timeReminderTime; }
  public void setTimeReminderTime(LocalTime timeReminderTime) { this.timeReminderTime = timeReminderTime; }
  public Integer getTimeReminderMinMinutes() { return timeReminderMinMinutes; }
  public void setTimeReminderMinMinutes(Integer timeReminderMinMinutes) { this.timeReminderMinMinutes = timeReminderMinMinutes; }

  /** Returns the minimum hours threshold, computed from minutes. Defaults to 4.0 if not set. */
  public double getTimeReminderMinHours() {
    return timeReminderMinMinutes != null ? timeReminderMinMinutes / 60.0 : 4.0;
  }

  /** Parses timeReminderDays CSV (e.g. "MON,TUE,WED,THU,FRI") into a Set of DayOfWeek. */
  public Set<DayOfWeek> getWorkingDays() {
    if (timeReminderDays == null || timeReminderDays.isBlank()) {
      return Collections.emptySet();
    }
    return Arrays.stream(timeReminderDays.split(","))
        .map(String::trim)
        .filter(s -> !s.isEmpty())
        .map(s -> DAY_ABBREVIATIONS.get(s.toUpperCase(Locale.ROOT)))
        .filter(Objects::nonNull)
        .collect(Collectors.toCollection(() -> EnumSet.noneOf(DayOfWeek.class)));
  }

  /** Updates all time reminder fields atomically. */
  public void updateTimeReminderSettings(
      boolean enabled, String days, LocalTime time, Integer minMinutes) {
    this.timeReminderEnabled = enabled;
    this.timeReminderDays = days;
    this.timeReminderTime = time;
    this.timeReminderMinMinutes = minMinutes;
  }
}
```

```java
  @Override
  public String assembleSystemPrompt(AiFirmProfile profile) {
    String systemTemplate = loadClasspathResource(SYSTEM_PROMPT_RESOURCE);
    String outputSchema = loadClasspathResource(OUTPUT_SCHEMA_RESOURCE);
    String profileBlock = firmProfileService.assembleProfileBlock();
    String riskCalibration =
        profile.getRiskCalibration() != null ? profile.getRiskCalibration() : "MODERATE";

  @Override
  public String assembleUserPrompt(SkillContext context) {
    UUID customerId = context.entityId();

    // Pre-flight: customer must exist
    Customer customer =
        customerRepository
            .findById(customerId)
            .orElseThrow(() -> new ResourceNotFoundException("Customer", customerId));

    // Pre-flight: customer must have uploaded documents
    List<Document> allDocs =
        documentRepository.findByScopeAndCustomerId(Document.Scope.CUSTOMER, customerId);
    List<Document> uploadedDocs =
        allDocs.stream().filter(d -> d.getStatus() == Document.Status.UPLOADED).toList();
    if (uploadedDocs.isEmpty()) {
      throw new InvalidStateException(
          "No documents", "Customer has no uploaded documents for FICA verification");
    }

    // Pre-flight: customer must have an active checklist with PENDING items
    List<ChecklistInstance> instances = checklistInstanceRepository.findByCustomerId(customerId);
    ChecklistInstance activeInstance =
        instances.stream()
            .filter(i -> "IN_PROGRESS".equals(i.getStatus()))
            .findFirst()
            .orElseThrow(

                        "No active checklist",
                        "Customer has no active compliance checklist with PENDING items"));

        checklistInstanceItemRepository.findByInstanceIdOrderBySortOrder(activeInstance.getId());
    List<ChecklistInstanceItem> pendingItems =
        allItems.stream().filter(i -> "PENDING".equals(i.getStatus())).toList();
    if (pendingItems.isEmpty()) {
      throw new InvalidStateException(
          "No active checklist", "Customer has no active compliance checklist with PENDING items");

    // Read documents (text extraction + vision fallback)
    FicaDocumentReader.DocumentReadResult readResult =
        ficaDocumentReader.readDocuments(uploadedDocs);

    // Assemble user prompt

    prompt.append("<customer>\n");

        customer.getCustomFields().getOrDefault("acct_entity_type", "INDIVIDUAL").toString();
    prompt.append("Type: ").append(entityType).append("\n");
    prompt.append("</customer>\n\n");

    prompt.append("<checklist template=\"").append(activeInstance.getTemplateId()).append("\">\n");
    for (ChecklistInstanceItem item : allItems) {
      prompt
          .append("- [")
          .append(item.getStatus())
          .append("] ")
          .append(item.getName())
          .append(" (requires document: ")
          .append(item.isRequiresDocument())
          .append(") id=")
          .append(item.getId())
          .append("\n");
    }
    prompt.append("</checklist>\n\n");

    prompt.append("<documents>\n");
    prompt.append(readResult.textContent());
    prompt.append("</documents>\n\n");

    prompt.append(
        "Review the uploaded documents against the FICA checklist items listed above. "
            + "For each checklist item, assess whether the uploaded documents satisfy the requirement. "
            + "Produce your response as valid JSON.");

        llmJsonParser.parse(objectMapper, outputContent, FicaVerificationOutput.class);

                execution,
                "MARK_KYC_COMPLETE",
                proposedAction,
                action.reasoning(),

        gates.add(gate);
      }
      // REQUEST_ADDITIONAL_DOCUMENT actions are informational — no gate created
    }

  }

    return false;

      if (is == null) {
        throw new IllegalStateException("Classpath resource not found: " + path);
      }
      return new String(is.readAllBytes(), StandardCharsets.UTF_8);
    } catch (IOException e) {
      throw new IllegalStateException("Failed to load classpath resource: " + path, e);
    }
  }
```

```java
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;

@Service
public class InvoiceEmailService {

  private static final Logger log = LoggerFactory.getLogger(InvoiceEmailService.class);
  private static final String TEMPLATE_NAME = "invoice-delivery";
  private static final String REFERENCE_TYPE = "INVOICE";

  private final IntegrationRegistry integrationRegistry;
  private final EmailTemplateRenderer emailTemplateRenderer;
  private final EmailContextBuilder emailContextBuilder;
  private final EmailDeliveryLogService deliveryLogService;
  private final EmailRateLimiter emailRateLimiter;
  private final String portalBaseUrl;

  public InvoiceEmailService(
      IntegrationRegistry integrationRegistry,
      EmailTemplateRenderer emailTemplateRenderer,
      EmailContextBuilder emailContextBuilder,
      EmailDeliveryLogService deliveryLogService,
      EmailRateLimiter emailRateLimiter,
      @Value("${docteams.app.portal-base-url:http://localhost:3002}") String portalBaseUrl) {
    this.integrationRegistry = integrationRegistry;
    this.emailTemplateRenderer = emailTemplateRenderer;
    this.emailContextBuilder = emailContextBuilder;
    this.deliveryLogService = deliveryLogService;
    this.emailRateLimiter = emailRateLimiter;
    this.portalBaseUrl = portalBaseUrl;
  }

  public void sendInvoiceEmail(Invoice invoice, byte[] pdfBytes) {
    String recipientEmail = invoice.getCustomerEmail();
    if (recipientEmail == null || recipientEmail.isBlank()) {
      log.warn("Skipping invoice email for invoice {} -- no customer email", invoice.getId());
      return;
    }

    if (!RequestScopes.TENANT_ID.isBound()) {
      log.warn("Skipping invoice email for invoice {} -- no tenant context", invoice.getId());
      return;
    }

    try {
      // 1. Resolve provider
      EmailProvider provider =
          integrationRegistry.resolve(IntegrationDomain.EMAIL, EmailProvider.class);

      // 2. Build context (base + invoice-specific)
      Map<String, Object> context =
          emailContextBuilder.buildBaseContext(invoice.getCustomerName(), null);
      context.put("customerName", invoice.getCustomerName());
      context.put("invoiceNumber", invoice.getInvoiceNumber());
      context.put("amount", invoice.getTotal().toPlainString());
      context.put("currency", invoice.getCurrency());
      context.put("dueDate", invoice.getDueDate() != null ? invoice.getDueDate().toString() : null);
      String orgName = (String) context.get("orgName");
      // GAP-L-65 -- subject uses the firm's terminology (e.g. "Fee Note INV-... from Acme Law"
      // for legal-za tenants). invoiceTerm always falls back to "Invoice" via EmailContextBuilder.
      String invoiceTerm = (String) context.get("invoiceTerm");
      context.put("subject", invoiceTerm + " " + invoice.getInvoiceNumber() + " from " + orgName);
      // GAP-L-64 — always populate portalUrl so the email "View Invoice" CTA renders with a
      // working href even when no PSP is configured (paymentUrl is null).
      context.put("portalUrl", portalBaseUrl + "/invoices/" + invoice.getId());
      context.put("paymentUrl", invoice.getPaymentUrl());

      // 3. Render template
      var rendered = emailTemplateRenderer.render(TEMPLATE_NAME, context);

      // 4. Check rate limit
      String tenantSchema = RequestScopes.TENANT_ID.get();
      if (!emailRateLimiter.tryAcquire(tenantSchema, provider.providerId())) {
        log.warn("Rate limit exceeded for invoice email, invoice={}", invoice.getId());
        deliveryLogService.recordRateLimited(
            REFERENCE_TYPE, invoice.getId(), TEMPLATE_NAME, recipientEmail, provider.providerId());
        return;
      }

      // 5. Construct message
      var message =
          EmailMessage.withTracking(
              recipientEmail,
              rendered.subject(),
              rendered.htmlBody(),
              rendered.plainTextBody(),
              null,
              REFERENCE_TYPE,
              invoice.getId().toString(),
              tenantSchema);

      // 6/7. Send — with the rendered PDF attached when one is available.
      var result =
          pdfBytes == null
              ? provider.sendEmail(message)
              : provider.sendEmailWithAttachment(
                  message,
                  new EmailAttachment(
                      invoice.getInvoiceNumber() + ".pdf", "application/pdf", pdfBytes));

      // 8. Record delivery
      deliveryLogService.record(
          REFERENCE_TYPE,
          invoice.getId(),
          TEMPLATE_NAME,
          recipientEmail,
          provider.providerId(),
          result);

      if (result.success()) {
        log.info("Invoice email sent for invoice={} to={}", invoice.getId(), recipientEmail);
      } else {
        log.warn("Invoice email failed for invoice={}: {}", invoice.getId(), result.errorMessage());
      }
    } catch (Exception e) {
      log.error("Unexpected error sending invoice email for invoice={}", invoice.getId(), e);
    }
  }
}
```

```java
  @Transactional(propagation = Propagation.REQUIRES_NEW)
  public Notification createNotification(  // unconditional variant
      UUID recipientMemberId, String type, String title, String body,
      String refEntityType, UUID refEntityId, UUID refProjectId)
```

Plus `notifyAdminsAndOwners(...)` fan-out helper, and event-driven `handleXxx(XxxEvent)` methods (e.g. `handleTaskAssigned(TaskAssignedEvent)`) invoked from `notification/NotificationEventHandler` which `@TransactionalEventListener`s the DomainEvents. So a new type = (1) string constant added to `NOTIFICATION_TYPES`, (2) a `handleXxx` / direct `createIfEnabled` call site, usually wired from a new DomainEvent via `NotificationEventHandler`.

### 9l. Existing AI-Gate Review Frontend Surface

Route: **`frontend/app/(app)/org/[slug]/ai/reviews/`** — three files:

- `page.tsx` — RSC. Guards on capabilities (`AI_REVIEW` or admin/owner via `fetchMyCapabilities()`); fetches gates with `getAiGates({ status, gateType, page, size })` from `lib/api/ai.ts`; for `CREATE_TASK_FROM_CORRESPONDENCE` gates it fetches each gate's detail (`getAiGate(id)`) to resolve the originating correspondence link (frontend-only enrichment, `Promise.allSettled`, best-effort); passes gate lists to the client component.
- `reviews-client.tsx` — client component; renders each gate through `<ExecutionGateCard …>` (`components/ai/execution-gate-card.tsx`), split into pending/history sections.
- `actions.ts` — server actions wrapping `approveAiGate(id, notes?)` → `POST /api/ai/gates/{id}/approve` and `rejectAiGate(id, notes?)` → `POST /api/ai/gates/{id}/reject`.

API client (`frontend/lib/api/ai.ts`): `AiGateListItem` (list DTO — id, gateType, status, aiReasoning, expiresAt; **no proposedAction**), `AiGateDetail` (adds `proposedAction`), `PaginatedGatesResponse`; functions `getAiGates(params)` → `GET /api/ai/gates?status=&gateType=&page=&size=`, `getAiGate(id)`, `approveAiGate(id, notes?)`, `rejectAiGate(id, notes?)`.

`ExecutionGateCard` is reused embedded in `components/ai/contract-review-button.tsx`, `components/ai/matter-intake-panel.tsx`, `components/ai/fica-verification-panel.tsx` — so gate approve/reject UX is a single shared card component. **Batch-approve implication**: today every approve is a single `POST /api/ai/gates/{id}/approve`; the list page approves one card at a time; there is no multi-select or bulk endpoint. A batch-approve extension would add a bulk action on `ai/reviews` + a new backend endpoint (or a loop over the existing one) on `AiExecutionGateController` (`integration/ai/gate/AiExecutionGateController.java`), with `AiExecutionGateService` → `GateActionExecutor.execute(gate)` per gate.

## Cross-Cutting Constraints Worth Restating for the Architecture Doc

- Schema-per-tenant: no `tenant_id` columns, no `@Filter`; tenant bound via `RequestScopes.TENANT_ID` ScopedValue → Hibernate search_path. Java 25, Spring Boot 4.0.2, Hibernate 7, no Lombok.
- Controllers are pure delegation (one service call, `ResponseEntity`, `@RequiresCapability`); services throw semantic exceptions (`ResourceNotFoundException` = 404 and doubles as access-denied security-by-obscurity).
- Two Jackson generations coexist: job-queue payloads use `com.fasterxml.jackson.databind.JsonNode`; AI skills use `tools.jackson.databind.ObjectMapper`.
- Per-tenant background work = `JobHandler` (job queue, V24 global) — not raw `@Scheduled` (ShedLock/sharding handled by JobWorker).
- New MCP tools (if any) trip `McpReadOnlyRegistryTest`: write tools need `WRITE_TOOL_NAMES`, read tools not named `list/get/search` need `READ_NAME_EXEMPTIONS`, new tool beans need `ALLOWED_TOOL_BEANS`.
- `AuditEventTypeRegistry` has a count assertion (currently 36 entries); `AiExecutionGate.status` is a free String, not an enum; there is no `AiGatePendingEvent` (only Approved/Rejected/Expired).
- `OrgSettingsSchemaSnapshotTest` pins the `org_settings` table shape — new settings columns need a migration + deliberate pin update; all-nullable embedded groups need `OrgSettingsEmbeddableNullReloadTest` coverage.
